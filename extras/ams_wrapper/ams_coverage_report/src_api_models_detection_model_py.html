28class DetectionModel(Model): 

29 def postprocess_inference_output(self, inference_output: dict) -> str: 

30 # Assuming single output 

31 output_config = next(iter(self.output_configs.values())) 

32 output_name = output_config.output_name 

33 result_array = inference_output[output_name] 

34 

35 # model with output shape (1,1,200,7)  

36 # with last dimension containg detection details 

37 detections = [] 

38 for detection in result_array[0][0]: 

39 label = detection[output_config.value_index_mapping['value']].item() 

40 # End of detections 

41 if label == 0.0: 

42 break 

43 

44 if not label in self.labels[output_name]: 

45 raise ValueError("label not found in labels definition") 

46 else: 

47 label_value = self.labels[output_name][label] 

48 

49 conf = detection[output_config.value_index_mapping['confidence']].item() 

50 x_min = detection[output_config.value_index_mapping['x_min']].item() 

51 y_min = detection[output_config.value_index_mapping['y_min']].item() 

52 x_max = detection[output_config.value_index_mapping['x_max']].item() 

53 y_max = detection[output_config.value_index_mapping['y_max']].item() 

54 

55 tag = Tag(label_value, conf) 

56 

57 box = Rectangle(x_min, y_min, abs(x_max-x_min), abs(y_max-y_min)) 

58 

59 detection = SingleEntity(tag, box) 

60 detections.append(detection) 

61 

62 if len(detections) == 0: 

63 response = None 

64 else: 

65 entity = Entity(subtype_name=self.model_name, entities=detections) 

66 response = json.dumps(entity.as_dict()) 

67 

68 return response
