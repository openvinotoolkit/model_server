41def preprocess_binary_image(image: bytes, channels: int = None, 

42 target_size: Tuple[int, int] = None, 

43 channels_first=True, 

44 dtype=tf.dtypes.float32, scale: float = None, 

45 standardization=False, 

46 reverse_input_channels=True) -> np.ndarray: 

47 """ 

48 Preprocess binary image in PNG, JPG or BMP format, producing numpy array as a result. 

49 

50 :param image: Image bytes 

51 :param channels: Number of image's channels 

52 :param target_size: A tuple of desired height and width 

53 :param channels_first: If set to True, image array will be in NCHW format, 

54 NHWC format will be used otherwise 

55 :param dtype: Data type that will be used for decoding 

56 :param scale: If passed, decoded image array will be multiplied by this value 

57 :param standardization: If set to true, image array values will be standarized 

58 to have mean 0 and standard deviation of 1 

59 :param reverse_input_channels: If set to True, image channels will be reversed 

60 from RGB to BGR format 

61 :raises TypeError: if type of provided parameters are incorrect 

62 :raises ValueError: if values of provided parameters is incorrect 

63 :raises ImageDecodeError(ValueError): if image cannot be decoded 

64 :raises ImageResizeError(ValueError): if image cannot be resized 

65 :raises ImageTransformError(ValueError): if image cannot be properly transformed 

66 :returns: Preprocessed image as numpy array 

67 """ 

68 

69 params_to_check = {'channels': channels, 

70 'scale': scale} 

71 

72 for param_name, value in params_to_check.items(): 

73 try: 

74 if value is not None and value < 0: 

75 raise ValueError('Invalid value {} for parameter {}.'.format(value, param_name)) 

76 except TypeError: 

77 raise TypeError('Invalid type {} for parameter {}.'.format(type(value), param_name)) 

78 

79 try: 

80 if target_size: 

81 height, width = target_size 

82 if height <= 0 or width <= 0: 

83 raise ValueError('Invalid target size parameter.') 

84 except TypeError: 

85 raise TypeError('Invalid target size type.') 

86 

87 if not isinstance(dtype, tf.dtypes.DType): 

88 raise TypeError('Invalid type {} for parameter dtype.'.format(type(dtype))) 

89 

90 try: 

91 decoded_image = tf.io.decode_image(image, channels=channels) 

92 decoded_image = tf.dtypes.cast(decoded_image, dtype) 

93 except Exception as e: 

94 raise ImageDecodeError('Provided image is invalid, unable to decode.') from e 

95 

96 if target_size: 

97 try: 

98 decoded_image = tf.image.resize(decoded_image, target_size) 

99 except Exception as e: 

100 raise ImageResizeError('Failed to resize provided binary image from: {} ' 

101 'to: {}.'.format(tf.shape(decoded_image), target_size)) from e 

102 

103 try: 

104 if standardization: 

105 decoded_image = tf.image.per_image_standardization(decoded_image) 

106 image_array = decoded_image.numpy() 

107 if reverse_input_channels: 

108 image_array = image_array[..., ::-1] 

109 if channels_first: 

110 image_array = np.transpose(image_array, [2, 0, 1]) 

111 if scale: 

112 array_type = image_array.dtype 

113 image_array = image_array * scale 

114 image_array = image_array.astype(array_type) 

115 except Exception as e: 

116 log.exception(str(e)) 

117 raise ImageTransformError('Failed to preprocess image, ' 

118 'check if provided parameters are correct.') from e 

119 

120 return image_array
