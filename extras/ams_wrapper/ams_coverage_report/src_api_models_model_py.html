48 def preprocess_binary_image(self, binary_image: bytes) -> np.ndarray: 

49 try: 

50 # Assuming single input for now 

51 preprocessing_config = next(iter(self.input_configs.values())) 

52 preprocessing_config = preprocessing_config.as_preprocessing_options() 

53 preprocessed_image = default_preprocessing(binary_image, **preprocessing_config) 

54 preprocessed_image = np.expand_dims(preprocessed_image, axis=0) 

55 except Exception as e:

65 def on_post(self, req, resp): 

66 # Main flow for the inference request 

67 

68 # Retrieve request headers as python dictionary 

69 request_headers = req.headers 

70 logger.debug(f"Received request with headers: {request_headers}") 

71 # Retrieve raw bytes from the request 

72 request_body = req.bounded_stream.read() 

73 

74 # Preprocess request body 

75 try: 

76 preprocessing_start_time = datetime.datetime.now() 

77 input_image = self.preprocess_binary_image(request_body) 

78 duration = (datetime.datetime.now() - 

79 preprocessing_start_time).total_seconds() * 1000 

80 logger.debug(f"Input preprocessing time: {duration} ms") 

81 except Exception as ex: 

82 logger.exception("Failed request preprocessing") 

83 body = {"message": str(ex)} 

84 resp.status = falcon.HTTP_400 

85 resp.body = json.dumps(body) 

86 return 

87 

88 # Send inference request to corresponding model in OVMS 

89 try: 

90 connection_start_time = datetime.datetime.now() 

91 inference_output = self.ovms_connector.send(input_image) 

92 duration = (datetime.datetime.now() - 

93 connection_start_time).total_seconds() * 1000 

94 logger.debug(f"OVMS request handling time: {duration} ms") 

95 except (ValueError, TypeError) as ex: 

96 logger.exception("Invalid request data") 

97 body = {"message": str(ex)} 

98 resp.status = falcon.HTTP_400 

99 resp.body = json.dumps(body) 

100 return 

101 except ModelNotFoundError as ex: 

102 logger.exception("Model not found") 

103 body = {"message": str(ex)} 

104 resp.status = falcon.HTTP_500 

105 resp.body = json.dumps(body) 

106 return 

107 except OvmsUnavailableError as ex: 

108 logger.exception("OVMS unavailable") 

109 body = {"message": str(ex)} 

110 resp.status = falcon.HTTP_503 

111 resp.body = json.dumps(body) 

112 return 

113 except Exception as ex: 

114 logger.exception("Internal OVMS error") 

115 body = {"message": str(ex)} 

116 resp.status = falcon.HTTP_500 

117 resp.body = json.dumps(body) 

118 return 

119 

120 # Postprocess 

121 try: 

122 postprocessing_start_time = datetime.datetime.now() 

123 results = self.postprocess_inference_output(inference_output) 

124 duration = (datetime.datetime.now() -
